# Direct primary endpoint test
/tool fetch url=$APIURL http-method=post http-data=$SENDDATA http-header-field="Content-Type: application/x-www-form-urlencoded" check-certificate=no duration=8s output=none

# Backup endpoint is intentionally disabled on ROS 6.49 because Apps Script URL returns redirect.
# If primary direct works, continue with BLOCK D/E/F only.

# BLOCK A - quick checks

# BLOCK B - DNS resolve checks
:put ("RESOLVE worker=" . [:resolve "mikrotik-monitor-proxy.hetgraphic17.workers.dev"])

# BLOCK C - HTTPS transport checks (wait for prompt after each line)

:put "FETCH script done"

# BLOCK D - re-apply het_CONFIG fully (this fixes the broken paste state)
/system script set [find name="het_CONFIG"] source={
  :global APIURL "https://mikrotik-monitor-proxy.hetgraphic17.workers.dev";
  :global BACKUPURL "";
  :global TOKEN "MONITOR_TOKEN_2026";
  :global SITE "KANO";
  :global ROUTER "AMBARIYYA_GLOBAL";
  :global WANIF "ether1";
  :global VPNHOST "vpn.hetdubai.com";
  :if ([:typeof $SENDOK] = "nothing") do={ :global SENDOK 0; }
  :if ([:typeof $SENDDATA] = "nothing") do={ :global SENDDATA ""; }
  :log info "het_CONFIG_LOADED";
}

# BLOCK E - re-apply het_HTTP_SEND fully
/system script set [find name="het_HTTP_SEND"] source={
  :global APIURL;
  :global BACKUPURL;
  :global SENDDATA;
  :global SENDOK;
  :local H "Content-Type: application/x-www-form-urlencoded";
  :if ([:len $APIURL] = 0) do={ /system script run het_CONFIG; }
  :set SENDOK 0;
  :local payload $SENDDATA;
  :if ([:len $payload] = 0) do={ :log error "het_HTTP_SEND_DATA_EMPTY"; :return; }
  :do {
    /tool fetch url=$APIURL http-method=post http-data=$payload http-header-field=$H check-certificate=no duration=8s output=none;
    :set SENDOK 1;
    :log info "het_HTTP_SEND_PRIMARY_OK";
  } on-error={
    :set SENDOK 0;
    :log warning "het_HTTP_SEND_PRIMARY_FAIL";
  }
  :if (($SENDOK != 1) and ([:len $BACKUPURL] > 0)) do={
    :do {
      /tool fetch url=$BACKUPURL http-method=post http-data=$payload http-header-field=$H check-certificate=no duration=10s output=none;
      :set SENDOK 1;
      :log warning "het_HTTP_SEND_BACKUP_OK";
    } on-error={
      :set SENDOK 0;
      :log error "het_HTTP_SEND_BACKUP_FAIL";
    }
  }
  :if ($SENDOK != 1) do={ :log error "het_HTTP_SEND_FAIL"; }
}

# BLOCK F - live manual tests
/system script run het_CONFIG

# One-shot POST test with tiny payload (must return to prompt without failure)
